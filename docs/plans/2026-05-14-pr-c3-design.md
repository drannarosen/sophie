# PR-C3 — Key-insights + Figures + Misconceptions (design doc)

## Context

PR-C3 ships **three pedagogical roles in one PR** — the third and largest
instance of the ADR 0038 pedagogy index pattern. Per the
[PR-C3 overview](./2026-05-14-pr-c3-overview.md), it extends PR-C1
(definitions) + PR-C2 (equations) to key-insights, figures, and
misconceptions while shipping a meaningful infrastructure refactor
(`createPedagogyStore<T>` factory) and introducing the **two-tier figures
model** to handle the asset/usage distinction.

Source primitives extended (NOT replaced):
- `<Aside>` — gains AsideKind `"misconception"` (already has `"key-insight"`); both keep title optional
- `<Callout>` — gains CalloutVariant `"misconception"` alongside existing `"caution"`; gains optional `id?: string` prop (symmetric with Aside's `id?`)
- `<Figure>` (registry mode) — unchanged props; the figure registry types consolidate into `@sophie/core/schema/pedagogy-index.ts` per ADR 0001

New components:
- `<FigureRef name="...">` — React HoverCard inline cross-ref with thumbnail + caption popover
- `<ChapterKeyInsights />`, `<CourseKeyInsights />` Astro consumers
- `<ChapterFigures />`, `<CourseFigures />` Astro consumers
- `<ChapterMisconceptions />`, `<CourseMisconceptions />` Astro consumers

New routes: `/key-insights`, `/figures`, `/misconceptions` in the smoke target.

PR-C3 inherits all four ADR 0038 Revisions-section realities (globalThis
accumulator, virtual-module bypass, script-tag SSR→CSR transfer, HoverCard
primitive). No new infrastructure decisions; ONE meaningful refactor
(`createPedagogyStore<T>` factory + migrate PR-C1's definitions-store +
PR-C2's equations-store + new figures stores to use it).

The 10 brainstorm-locked decisions in the overview translate to an
~30-unit + ~15-e2e test budget across ~18 subagent tasks.

### `@sophie/components/src/runtime/pedagogy-store.ts` — NEW factory

```ts
/**
 * Factory that produces the SSR→CSR-aware role-store shape used by
 * `<GlossaryTerm>`, `<EqRef>`, `<FigureRef>` (and future inline-ref
 * components). Each role's store is a typed instance with:
 *
 *   - `set(entries)`: SSR-side setter (called by TextbookLayout)
 *   - `lookup(key)`: client-side lookup; auto-hydrates from a
 *     `<script id="...">` JSON payload on first call when no SSR
 *     setter ran
 *
 * Per ADR 0038 Revisions section: each role gets its own `<script>` tag
 * id; data flows server → __set*() OR script-tag-on-first-lookup → client.
 *
 * Extracts the boilerplate that PR-C1's definitions-store and PR-C2's
 * equations-store duplicated verbatim. PR-C3 migrates both to use this.
 */

export interface PedagogyStoreOptions<T> {
  /** DOM id of the SSR-emitted <script type="application/json"> tag. */
  scriptId: string;
  /** Tag prefix for dev-only console.error logs. e.g. "[EqRef]" or "[FigureRef:registry]". */
  logTag: string;
  /** Function that returns the lookup key for an entry (typically the slug or name). */
  keyOf: (entry: T) => string;
}

export interface PedagogyStore<T> {
  /** SSR setter: populate from server-side index. Suppresses later auto-hydration. */
  set: (entries: ReadonlyArray<T>) => void;
  /** Client-side lookup. Auto-hydrates from the script tag on first call when no setter ran. */
  lookup: (key: string) => T | undefined;
}

export function createPedagogyStore<T>(
  opts: PedagogyStoreOptions<T>
): PedagogyStore<T> {
  let byKey: Map<string, T> = new Map();
  let hydratedFromScript = false;

  function hydrateFromScriptTagIfPresent(): void {
    if (hydratedFromScript) return;
    if (typeof document === "undefined") return;
    const el = document.getElementById(opts.scriptId);
    const raw = el?.textContent;
    if (!raw) {
      hydratedFromScript = true;
      return;
    }
    try {
      const parsed = JSON.parse(raw);
      if (!Array.isArray(parsed)) {
        throw new TypeError(
          `Expected an array; got ${typeof parsed}`
        );
      }
      const entries = parsed as ReadonlyArray<T>;
      byKey = new Map(entries.map((e) => [opts.keyOf(e), e]));
    } catch (err) {
      if (
        typeof process === "undefined" ||
        process.env?.NODE_ENV !== "production"
      ) {
        console.error(
          `${opts.logTag} Failed to parse \`${opts.scriptId}\` script payload; lookups will return undefined for this page.`,
          err
        );
      }
    }
    hydratedFromScript = true;
  }

  return {
    set(entries) {
      hydratedFromScript = true;
      byKey = new Map(entries.map((e) => [opts.keyOf(e), e]));
    },
    lookup(key) {
      hydrateFromScriptTagIfPresent();
      return byKey.get(key);
    },
  };
}
```

```ts
import type { FigureUsageEntry } from "@sophie/core/schema";

const store = createPedagogyStore<FigureUsageEntry>({
  scriptId: "sophie-pedagogy-figure-usages",
  logTag: "[FigureRef:usages]",
  keyOf: (u) => `${u.chapter}#${u.name}`,  // unique per chapter+name
});

export const __setFigureUsages = store.set;
export const lookupFigureUsage = store.lookup;
// Also export a helper that returns the canonical usage for a given name:
export function lookupCanonicalFigureUsage(
  name: string,
  allUsages: ReadonlyArray<FigureUsageEntry>
): FigureUsageEntry | undefined {
  // Prefer explicit canonical; else first by chapter+number sort
  const matches = allUsages.filter((u) => u.name === name);
  if (matches.length === 0) return undefined;
  const explicit = matches.find((u) => u.canonical);
  if (explicit) return explicit;
  return matches
    .slice()
    .sort((a, b) =>
      a.chapter === b.chapter ? a.number - b.number : a.chapter.localeCompare(b.chapter)
    )[0];
}
```

(Note: `lookupCanonicalFigureUsage` needs all usages because the store
keys by `chapter#name`. We expose an `allUsages` accessor below.)

For the canonical lookup we'll also expose the FULL usages array via a
separate accessor that the FigureRef component can iterate. Refactor: the
`figure-usages-store` exports `__setFigureUsages`, `lookupFigureUsage` (by
composite key), AND `lookupCanonicalUsageByName(name)` (which iterates an
internal slice). Implementation hides the slice inside the module:

```ts
// figure-usages-store.ts (revised — single API, no caller-provided slice)
let allUsages: ReadonlyArray<FigureUsageEntry> = [];

const store = createPedagogyStore<FigureUsageEntry>({...});

export function __setFigureUsages(entries: ReadonlyArray<FigureUsageEntry>): void {
  allUsages = entries;
  store.set(entries);
}

export const lookupFigureUsage = store.lookup;  // by chapter#name

export function lookupCanonicalUsageByName(name: string): FigureUsageEntry | undefined {
  const matches = allUsages.filter((u) => u.name === name);
  if (matches.length === 0) return undefined;
  return (
    matches.find((u) => u.canonical) ??
    [...matches].sort((a, b) =>
      a.chapter === b.chapter ? a.number - b.number : a.chapter.localeCompare(b.chapter)
    )[0]
  );
}
```

### `@sophie/astro/src/lib/pedagogy-index-extractor.ts` — additions

Three new extractor functions (alongside existing `extractDefinitions` +
`extractEquations`):

```ts
import {
  type FigureUsageEntry,
  type KeyInsightEntry,
  type MisconceptionEntry,
  slugify,
} from "@sophie/core/schema";

// Helper attribute reader (generalized to handle key/value pairs)
interface AsideAttributes {
  kind?: string;
  title?: string;
  id?: string;
}
interface CalloutAttributes {
  variant?: string;
  title?: string;
  id?: string;  // NEW in PR-C3
}
interface FigureAttributes {
  name?: string;
  caption?: string;
  canonical?: boolean;
}

function readCalloutAttributes(node: MdxJsxFlowElement): CalloutAttributes {
  const out: CalloutAttributes = {};
  for (const attr of node.attributes ?? []) {
    if (attr.type !== "mdxJsxAttribute") continue;
    if (attr.name === "variant" && typeof attr.value === "string")
      out.variant = attr.value;
    if (attr.name === "title" && typeof attr.value === "string")
      out.title = attr.value;
    if (attr.name === "id" && typeof attr.value === "string")
      out.id = attr.value;
  }
  return out;
}

function readFigureAttributes(node: MdxJsxFlowElement): FigureAttributes {
  const out: FigureAttributes = {};
  for (const attr of node.attributes ?? []) {
    if (attr.type !== "mdxJsxAttribute") continue;
    if (attr.name === "name" && typeof attr.value === "string")
      out.name = attr.value;
    if (attr.name === "caption" && typeof attr.value === "string")
      out.caption = attr.value;
    // `canonical` JSX prop is a value-less boolean attribute (<Figure canonical />)
    // which the AST represents as `value: null`. Detect presence:
    if (attr.name === "canonical") {
      out.canonical = attr.value === null || attr.value === true;
    }
  }
  return out;
}

/**
 * Pure extractor for <Aside kind="key-insight"> blocks. Same shape
 * as extractDefinitions: walk the tree, extract, throw on
 * intra-chapter anchor collision. Title is optional; when absent,
 * anchor auto-generates as "key-insight-{n}" per chapter.
 */
export function extractKeyInsights(
  tree: Root,
  chapterSlug: string
): KeyInsightEntry[] {
  const out: KeyInsightEntry[] = [];
  const seenAnchors = new Set<string>();
  let counter = 0;

  visit(tree, "mdxJsxFlowElement", (node: unknown) => {
    const el = node as MdxJsxFlowElement;
    if (el.name !== "Aside") return;
    const attrs = readAsideAttributes(el);
    if (attrs.kind !== "key-insight") return;

    counter += 1;
    const titleSlug = attrs.title?.trim()
      ? slugify(attrs.title.trim())
      : null;
    const explicitId = attrs.id?.trim() ? slugify(attrs.id.trim()) : null;
    const anchor = explicitId ?? titleSlug ?? `key-insight-${counter}`;

    if (seenAnchors.has(anchor)) {
      throw new Error(
        `Intra-chapter anchor collision in chapter "${chapterSlug}": key-insight anchor "${anchor}" is generated by more than one <Aside kind="key-insight">. Resolution: add an explicit \`id\` prop to disambiguate.`
      );
    }
    seenAnchors.add(anchor);

    const body = renderChildrenToHtml(el.children);
    if (body.trim().length === 0) {
      // K2: empty-body WARNING — defense-in-depth as a console.warn
      // (not a throw; pre-launch authoring may legitimately have stub bodies)
      if (
        typeof process === "undefined" ||
        process.env?.NODE_ENV !== "production"
      ) {
        console.warn(
          `[pedagogy-index] <Aside kind="key-insight"> in chapter "${chapterSlug}" anchor "${anchor}" has empty body.`
        );
      }
    }

    out.push({
      title: attrs.title?.trim() || undefined,
      body,
      chapter: chapterSlug,
      anchor,
    });
  });

  return out;
}

/**
 * Pure extractor for <Figure name="..."> registry-mode blocks. Walks
 * the tree, captures per-chapter sequential `number`, detects intra-
 * chapter anchor collision (F5) and multiple-canonical-per-name (F3).
 *
 * Note: extractor does NOT verify the figure name is in the registry
 * (extractor lacks registry access; per Q10, F1 defers to PR-C4).
 */
export function extractFigures(
  tree: Root,
  chapterSlug: string
): FigureUsageEntry[] {
  const out: FigureUsageEntry[] = [];
  const seenAnchors = new Set<string>();
  let counter = 0;

  visit(tree, "mdxJsxFlowElement", (node: unknown) => {
    const el = node as MdxJsxFlowElement;
    if (el.name !== "Figure") return;
    const attrs = readFigureAttributes(el);
    if (!attrs.name) return;  // inline-mode <Figure src> doesn't index

    counter += 1;
    const name = attrs.name.trim();
    if (!name) return;

    const anchor = `fig-${slugify(name)}-${counter}`;

    if (seenAnchors.has(anchor)) {
      throw new Error(
        `Intra-chapter anchor collision in chapter "${chapterSlug}": figure anchor "${anchor}" generated by more than one <Figure name="${name}">. (F5 invariant.)`
      );
    }
    seenAnchors.add(anchor);

    out.push({
      name,
      chapter: chapterSlug,
      anchor,
      number: counter,
      canonical: attrs.canonical === true,
      captionOverride: attrs.caption?.trim() || undefined,
    });
  });

  return out;
}

/**
 * Pure extractor for misconceptions from BOTH source primitives.
 *
 *   - <Aside kind="misconception"> → length: "short"
 *   - <Callout variant="misconception"> → length: "long"
 *
 * Single function walks both; length discriminator from node.name.
 * Throws on intra-chapter anchor collision (M1) across BOTH sources.
 * Warns on empty body (M3).
 */
export function extractMisconceptions(
  tree: Root,
  chapterSlug: string
): MisconceptionEntry[] {
  const out: MisconceptionEntry[] = [];
  const seenAnchors = new Set<string>();
  let counter = 0;

  visit(tree, "mdxJsxFlowElement", (node: unknown) => {
    const el = node as MdxJsxFlowElement;
    let length: "short" | "long";
    let attrs: { title?: string; id?: string };

    if (el.name === "Aside") {
      const a = readAsideAttributes(el);
      if (a.kind !== "misconception") return;
      length = "short";
      attrs = { title: a.title, id: a.id };
    } else if (el.name === "Callout") {
      const c = readCalloutAttributes(el);
      if (c.variant !== "misconception") return;
      length = "long";
      attrs = { title: c.title, id: c.id };
    } else {
      return;
    }

    counter += 1;
    const titleSlug = attrs.title?.trim()
      ? slugify(attrs.title.trim())
      : null;
    const explicitId = attrs.id?.trim() ? slugify(attrs.id.trim()) : null;
    const anchor =
      explicitId ?? titleSlug ?? `misconception-${counter}`;

    if (seenAnchors.has(anchor)) {
      throw new Error(
        `Intra-chapter anchor collision in chapter "${chapterSlug}": misconception anchor "${anchor}" generated by more than one source. (M1 invariant.)`
      );
    }
    seenAnchors.add(anchor);

    const body = renderChildrenToHtml(el.children);
    if (body.trim().length === 0) {
      if (
        typeof process === "undefined" ||
        process.env?.NODE_ENV !== "production"
      ) {
        console.warn(
          `[pedagogy-index] misconception in chapter "${chapterSlug}" anchor "${anchor}" has empty body. (M3 warning.)`
        );
      }
    }

    out.push({
      body,
      chapter: chapterSlug,
      anchor,
      length,
      label: attrs.title?.trim() || undefined,
    });
  });

  return out;
}
```

### Accumulator extension

```ts
// Extend GlobalIndexState:
interface GlobalIndexState {
  definitions: Map<string, DefinitionEntry>;
  equations: Map<string, EquationEntry>;
  // NEW collections (PR-C3):
  keyInsights: Map<string, KeyInsightEntry>;
  figureUsages: Map<string, FigureUsageEntry>;
  misconceptions: Map<string, MisconceptionEntry>;
}

class IndexAccumulator {
  clearChapter(chapterSlug: string): void {
    const state = getGlobalState();
    for (const map of [
      state.definitions,
      state.equations,
      state.keyInsights,
      state.figureUsages,
      state.misconceptions,
    ]) {
      for (const [key, entry] of map) {
        if (entry.chapter === chapterSlug) map.delete(key);
      }
    }
  }

  addDefinitions(...): void { /* unchanged */ }
  addEquations(...): void { /* unchanged */ }

  /** NEW. Add key-insights for one chapter. Key: chapter#anchor (no cross-chapter collision check; key-insights are chapter-local). */
  addKeyInsights(entries: ReadonlyArray<KeyInsightEntry>): void {
    const state = getGlobalState();
    for (const entry of entries) {
      state.keyInsights.set(`${entry.chapter}#${entry.anchor}`, entry);
    }
  }

  /**
   * NEW. Add figure usages. F3 invariant: multiple canonical for same name
   * (across chapters) throws. Key: chapter#name (one usage per name per
   * chapter; if same chapter renders the same figure twice, that's a
   * different invariant — flagged for PR-C4 or future).
   */
  addFigureUsages(entries: ReadonlyArray<FigureUsageEntry>): void {
    const state = getGlobalState();
    // Two-pass: validate canonical uniqueness first
    for (const entry of entries) {
      if (entry.canonical) {
        for (const existing of state.figureUsages.values()) {
          if (existing.name === entry.name && existing.canonical) {
            throw new Error(
              `Multiple <Figure name="${entry.name}" canonical /> usages — found in "${existing.chapter}" and "${entry.chapter}". F3 invariant: exactly one canonical usage per figure name. Resolution: remove `canonical` from one.`
            );
          }
        }
      }
    }
    for (const entry of entries) {
      state.figureUsages.set(`${entry.chapter}#${entry.name}`, entry);
    }
  }

  /** NEW. Add misconceptions. M2: cross-chapter slug collision throws (mirrors definitions/equations pattern when explicit ids used). */
  addMisconceptions(entries: ReadonlyArray<MisconceptionEntry>): void {
    const state = getGlobalState();
    // Two-pass: cross-chapter slug collision check (only when the anchor
    // is an explicit id-derived slug, not the auto "misconception-{n}"
    // pattern which is naturally chapter-scoped)
    for (const entry of entries) {
      const key = `${entry.chapter}#${entry.anchor}`;
      for (const existing of state.misconceptions.values()) {
        if (
          existing.chapter !== entry.chapter &&
          existing.anchor === entry.anchor &&
          !entry.anchor.startsWith("misconception-")
        ) {
          throw new Error(
            `Misconception slug "${entry.anchor}" defined in multiple chapters: "${existing.chapter}" and "${entry.chapter}". (M2 invariant.) Resolution: change one of the \`id\` props.`
          );
        }
      }
    }
    for (const entry of entries) {
      state.misconceptions.set(`${entry.chapter}#${entry.anchor}`, entry);
    }
  }

  asPedagogyIndex(): PedagogyIndex {
    const state = getGlobalState();
    return {
      definitions: Array.from(state.definitions.values()),
      equations: Array.from(state.equations.values()),
      keyInsights: Array.from(state.keyInsights.values()),
      figureRegistry: [],  // populated by consumer SSR-merge, not the extractor
      figureUsages: Array.from(state.figureUsages.values()),
      misconceptions: Array.from(state.misconceptions.values()),
    };
  }
}
```
