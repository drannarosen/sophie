<SkillReview.Card id="product-rule">
  <SkillReview.Prompt>What does log_b(xy) equal?</SkillReview.Prompt>
  <SkillReview.Answer>log_b(x) + log_b(y)</SkillReview.Answer>
</SkillReview.Card>

<SkillReview.Card id="power-rule">
  <SkillReview.Prompt>What does log_b(x^n) equal?</SkillReview.Prompt>
  <SkillReview.Answer>n * log_b(x)</SkillReview.Answer>
</SkillReview.Card>

<SkillReview.Card id="change-of-base">
  <SkillReview.Prompt>Express log_2(x) using natural log.</SkillReview.Prompt>

</SkillReview.Card>
